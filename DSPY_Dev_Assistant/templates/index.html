A Self-Correcting Developer Tool

Explore the architecture of a powerful assistant built with the DSPy framework. It generates, explains, and fixes code while fact-checking its own output to ensure accuracy and reliability.

Core Features

üß†

RAG

The assistant uses Retrieval-Augmented Generation to pull information from a knowledge base, ensuring its responses are context-aware and well-informed.

üß±

Modular Architecture

Built with composable `dspy.Module` classes, the architecture is clean, scalable, and easy to maintain or extend with new capabilities.

‚úÖ

Self-Correction

A custom `VerificationModule` fact-checks generated code against documentation and automatically corrects it if necessary, ensuring high reliability.

Optimized Compilation

Uses DSPy's `BootstrapFewShot` compiler to learn from examples, significantly improving reliability and reducing code hallucinations.

How to Run

1

Install Dependencies

Get all the necessary packages for the project.

pip install -r requirements.txt
2

Configure Environment

Set your `OPENAI_API_KEY` environment variable. The assistant is pre-configured for OpenAI but can be adapted for any DSPy-supported LLM.

3

Run the Assistant

Execute the main script to initialize, compile, and run a sample query.

python main.py

import dspy
from pydantic import BaseModel, Field

# --- Signatures for Core Generation Tasks ---

class CodeGenerator(dspy.Signature):
    """Generate Python code based on a detailed description."""
    description = dspy.InputField(desc="A detailed description of the desired code's functionality.")
    code = dspy.OutputField(desc="The generated Python code.")

class CodeQuestionAnswering(dspy.Signature):
    """Answer a question about the given code context."""
    context = dspy.InputField(desc="Relevant code snippets from the user's codebase.")
    question = dspy.InputField(desc="The user's question about the code.")
    answer = dspy.OutputField(desc="A detailed answer to the question.")

# --- Signature for the Verification Module ---

class VerificationSignature(dspy.Signature):
    """Verify a generated answer against the provided documentation context. If the answer is incorrect, correct it."""
    context = dspy.InputField(desc="Relevant documentation for fact-checking.")
    question = dspy.InputField(desc="The original user question or programming task.")
    generated_answer = dspy.InputField(desc="The answer produced by the assistant that needs to be checked.")

    verdict = dspy.OutputField(desc="A simple 'Correct' or 'Incorrect' verdict based on the context.")
    corrected_answer = dspy.OutputField(desc="The corrected answer. If the verdict is 'Correct', this should be the same as the generated_answer.")
                    

import dspy
from .signatures import CodeGenerator, VerificationSignature

class VerificationModule(dspy.Module):
    """A module to verify and correct generated code."""
    def __init__(self):
        super().__init__()
        self.retriever = dspy.ColBERTv2(url='http://20.102.90.50:2017/wiki17_abstracts')
        self.verifier = dspy.Predict(VerificationSignature)

    def forward(self, question, generated_answer):
        context = self.retriever(question, k=1).passages[0]
        result = self.verifier(
            context=context,
            question=question,
            generated_answer=generated_answer
        )
        return result

class DevAssistant(dspy.Module):
    """The main module for our self-correcting developer assistant."""
    def __init__(self):
        super().__init__()
        self.generate_code = dspy.Predict(CodeGenerator)
        self.verifier = VerificationModule()

    def forward(self, user_query, task="generate"):
        if task == "generate":
            generated_output = self.generate_code(description=user_query)
            verification_result = self.verifier(
                question=user_query,
                generated_answer=generated_output.code
            )
            if "incorrect" in verification_result.verdict.lower():
                return verification_result.corrected_answer
            else:
                return generated_output.code
        else:
            raise ValueError(f"Unknown task: {task}")
                    

import dspy

def get_trainset():
    """Returns a list of training examples for compiling the DevAssistant."""
    train_example_1 = dspy.Example(
        question="Write a Python loop to iterate over a dictionary's items.",
        generated_answer="for key, value in my_dict.iteritems(): print(f'{key}: {value}')",
        context="In Python 3, the `iteritems()` method was removed. You should use the `items()` method instead.",
        verdict="Incorrect",
        corrected_answer="for key, value in my_dict.items(): print(f'{key}: {value}')"
    ).with_inputs("question", "generated_answer", "context")

    train_example_2 = dspy.Example(
        question="How do I open a file in Python?",
        generated_answer="with open('my_file.txt', 'r') as f: content = f.read()",
        context="The recommended way to open a file in Python is using the `with` statement.",
        verdict="Correct",
        corrected_answer="with open('my_file.txt', 'r') as f: content = f.read()"
    ).with_inputs("question", "generated_answer", "context")

    return [train_example_1, train_example_2]
                    

import dspy
import os
from assistant.modules import DevAssistant
from data.trainset import get_trainset

def main():
    """Main function to configure DSPy, compile, and run the DevAssistant."""
    llm = dspy.OpenAI(model='gpt-4o-mini', max_tokens=4000)
    dspy.settings.configure(lm=llm)

    trainset = get_trainset()

    compiler = dspy.BootstrapFewShot(metric=None)

    print("INFO: Compiling the DevAssistant...")
    compiled_assistant = compiler.compile(DevAssistant(), trainset=trainset)
    print("INFO: Compilation complete.")

    user_query = "Write a Python loop to iterate over a dictionary's items."
    final_code = compiled_assistant(user_query=user_query)

    print("\n--- Final Generated Code ---")
    print(final_code)
    print("--------------------------\n")

if __name__ == "__main__":
    main()
